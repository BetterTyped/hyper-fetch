## useSubmit

For submitting and mutating data on server use the `useSubmit` hook that automates command handling. There are tons of
options to adjust this hook so check out the documentation.

```tsx
import { postUser, patchUser, deleteUser } from "server/user";

...

const { submit: submitPost, isSubmitting } = useSubmit(postUser)

const { submit: submitPatch, isSubmitting } = useSubmit(patchUser.setParams({ userId: 1 }))

const { submit: submitDelete, isSubmitting: isSubmittingDelete } = useSubmit(patchUser)


...


const handlePost = (values: { email: string, password: string }) => {
  submitPost({ data: values })
}

const handlePatch = (values: { email: string }) => {
  submitPatch({ data: values })
}



return (
  <button type="button" onClick={() => submitDelete({ params: { userId: 1 } })} disabled={isSubmittingDelete}>Delete</button>
)


```
