# Submitting

> For submitting and mutating data on server use the
> [useSubmit](https://hyperfetch.bettertyped.com/docs/documentation/react/core/use-submit/) hook that automates request
> handling.

---

```tsx
import { useSubmit } from "@hyper-fetch/react";
import { postUser } from "./requests";

const App = () => {
  // highlight-next-line
  const { submit, submitting } = useSubmit(postUser);

  const handleSubmit = async (
    values: { email: string; password: string },
    { resetForm }: { resetForm: VoidFunction },
  ) => {
    // We can await for response from submit method since it returns a promise with response data!
    // highlight-next-line
    const response = await submit({ data: values }); // Pass dynamic data values to submit
    resetForm();
  };

  return (
    <Form onSubmit={handleSubmit}>
      <Field name="email" type="email" />
      <Field name="password" type="password" />
      <button type="submit" disabled={submitting}>
        Submit
      </button>
    </Form>
  );
};
```

##### Read more about [useSubmit and submitting](https://hyperfetch.bettertyped.com/docs/documentation/react/core/use-submit/) on Hyper Fetch documentation.
